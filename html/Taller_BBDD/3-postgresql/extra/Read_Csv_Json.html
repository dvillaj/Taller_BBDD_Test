import pandas as pd
from io import StringIO
import requests

url="http://datos.madrid.es/egob/catalogo/212411-16-madrid-avisa.csv"
data=StringIO(requests.get(url).text)

df = pd.read_csv(data, delimiter=";", encoding = "ISO-8859-1")

# Take a sample
df = df.sample(1000)
